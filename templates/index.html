        Code Reference & Documentation

import azure.cognitiveservices.speech as speechsdk

speech_config = speechsdk.SpeechConfig(subscription="KEY", region="REGION")
speech_config.speech_synthesis_voice_name = "zh-CN-XiaoxiaoNeural"
synthesizer = speechsdk.SpeechSynthesizer(speech_config=speech_config)

# Plain text
result = synthesizer.speak_text_async("你好，世界！").get()

# SSML
ssml = """<speak version="1.0" xmlns="http://www.w3.org/2001/10/synthesis" xml:lang="zh-CN">
  <voice name="zh-CN-XiaoxiaoNeural">你好，世界！</voice>
</speak>"""
result = synthesizer.speak_ssml_async(ssml).get()

Speech Translation

      

        
        
      


      
      

        
Real-time speech translation using Azure Speech SDK TranslationRecognizer.

        

          Code Reference & Documentation

import azure.cognitiveservices.speech as speechsdk

translation_config = speechsdk.translation.SpeechTranslationConfig(
    subscription="KEY", region="REGION",
    speech_recognition_language="zh-CN",
    target_languages=["en", "ja"])
audio_config = speechsdk.audio.AudioConfig(use_default_microphone=True)

recognizer = speechsdk.translation.TranslationRecognizer(
    translation_config=translation_config, audio_config=audio_config)

def on_recognized(evt):
    print(f"Original: {evt.result.text}")
    for lang, text in evt.result.translations.items():
        print(f"  [{lang}]: {text}")

recognizer.recognized.connect(on_recognized)
recognizer.start_continuous_recognition()
input("Press Enter to stop...\n")
recognizer.stop_continuous_recognition()

Real-time speech-to-speech translation with automatic language detection.

        

          Code Reference & Documentation

import azure.cognitiveservices.speech as speechsdk

endpoint = "wss://{region}.stt.speech.microsoft.com/speech/universal/v2"
translation_config = speechsdk.translation.SpeechTranslationConfig(
    endpoint=endpoint, subscription="KEY")
translation_config.add_target_language("en")
translation_config.voice_name = "en-US-AvaMultilingualNeural"  # or "personal-voice"

auto_detect = speechsdk.languageconfig.AutoDetectSourceLanguageConfig()
audio_config = speechsdk.audio.AudioConfig(use_default_microphone=True)

recognizer = speechsdk.translation.TranslationRecognizer(
    translation_config=translation_config,
    audio_config=audio_config,
    auto_detect_source_language_config=auto_detect)

recognizer.recognized.connect(lambda evt: print(evt.result.translations))
recognizer.synthesizing.connect(lambda evt: print(f"Audio: {len(evt.result.audio)} bytes"))
recognizer.start_continuous_recognition()

Voice Live

      
Real-time voice conversation with AI using Azure Voice Live API.

      

        Code Reference & Documentation

import asyncio
from azure.core.credentials import AzureKeyCredential
from azure.ai.voicelive.aio import connect
from azure.ai.voicelive.models import (
    Modality, RequestSession, ServerVad, ServerEventType,
    InputAudioFormat, OutputAudioFormat,
)

async def main():
    credential = AzureKeyCredential("KEY")
    async with connect(
        endpoint="ENDPOINT", credential=credential, model="gpt-4o-mini"
    ) as conn:
        await conn.session.update(session=RequestSession(
            modalities=[Modality.TEXT, Modality.AUDIO],
            instructions="You are a helpful assistant.",
            voice="alloy",
            input_audio_format=InputAudioFormat.PCM16,
            output_audio_format=OutputAudioFormat.PCM16,
            turn_detection=ServerVad(),
        ))
        async for event in conn:
            if event.type == ServerEventType.RESPONSE_AUDIO_TRANSCRIPT_DELTA:
                print(event.delta, end="", flush=True)

asyncio.run(main())

import requests, json

url = "https://{region}.api.cognitive.microsoft.com/speechtotext/transcriptions:transcribe?api-version=2025-10-15"
headers = {"Ocp-Apim-Subscription-Key": "KEY"}

# Transcribe (set task to "translate" for translation)
definition = {
    "enhancedMode": {
        "enabled": True,
        "task": "transcribe",           # or "translate"
        # "targetLanguage": "en",       # required for translate
        # "prompt": ["Output in lexical format."],
    }
}

with open("audio.wav", "rb") as f:
    files = {
        "audio": ("audio.wav", f),
        "definition": (None, json.dumps(definition), "application/json"),
    }
    resp = requests.post(url, headers=headers, files=files)

print(resp.json()["combinedPhrases"][0]["text"])
